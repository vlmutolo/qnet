\begin{abstract}
We consider a quantum network where Bell pairs $[x,y]$ can be generated on certain links and can be moved across the network via entanglement swapping. Each node $i$ can perform at most $K_i$ swaps per time slot. Consumption requests arrive between end nodes $(x,y)$ at rates $\lambda_{xy}$, and must be served by consuming Bell pairs $[x,y]$.

We incorporate a distillation cost parameter $B\in\mathbb{Z}_{\ge 1}$: each logical consumption request and each entanglement swap consumes $B$ physical Bell pairs on every input link, while producing one Bell pair on the output link. We propose a fully distributed backpressure rule: at each node, schedule up to $K_i$ swaps with the largest positive ``pressure differences'' $W_i(y,y';t)$, and serve demand on pair $xy$ if $\gamma_{xy}(t)\ge B\alpha_{xy}(t)$. We prove throughput optimality using Lyapunov drift. The proof is structured using two dual queues: $\gamma_{xy}(t)$ (demand pressure) and $\alpha_{xy}(t)$ (resource scarcity pressure). We show (i) how these queues arise from the partial dual of a feasibility LP with distillation, (ii) why the per-slot greedy rule is a primal maximizer given these duals, and (iii) how a quadratic Lyapunov on $(\gamma,\alpha)$ certifies stability under any offered load $\lambda$ strictly inside the capacity region $\Lambda^{K,B}$.
\end{abstract}

\section{Network model}



Let $\Ncal$ be a finite node set. Let
\[
\Pairs \;:=\; \{\, xy : x,y\in\Ncal,\ x<y \,\}
\]
denote unordered distinct node pairs (we write $xy$ for $(x,y)$). For each $xy\in\Pairs$:

\begin{itemize}[leftmargin=2em]
\item $A_{xy}(t)\in\mathbb{Z}_{\ge 0}$ is the number of consumption \emph{requests} arriving in slot $t$ that require consuming Bell pairs $[x,y]$. We assume $\E[A_{xy}(t)]=\lambda_{xy}$ and that $\{A_{xy}(t)\}$ have uniformly bounded second moments.
\item $G_{xy}(t)\in\mathbb{Z}_{\ge 0}$ is the number of Bell pairs $[x,y]$ \emph{directly generated} in slot $t$ (e.g., via physical entanglement generation on a link). We assume $\E[G_{xy}(t)]=g_{xy}$, and $\{G_{xy}(t)\}$ have uniformly bounded second moments.
\end{itemize}

We assume $G_{xy}(t)=G_{yx}(t)$ and $A_{xy}(t)=A_{yx}(t)$, and for all $x$, $G_{xx}(t)=A_{xx}(t)=0$ (no self-pairs).

\paragraph{Distillation cost.}

Each \emph{logical} consumption request and each \emph{logical} entanglement swap consumes $B$ physical Bell pairs on its input links and produces one Bell pair on the output link. $B\in\mathbb{Z}_{\ge 1}$ is a constant.

\subsection{Queue of Bell pairs}

Let $Q_{xy}(t)\in\mathbb{Z}_{\ge 0}$ denote the number of Bell pairs currently available between $x$ and $y$ at the \emph{beginning} of slot $t$. 

During slot $t$, we will:
\begin{enumerate}[leftmargin=2em,label=(\alph*)]
\item generate new pairs $G_{xy}(t)$,
\item perform entanglement swaps,
\item and then consume pairs either to satisfy consumption demand or as swap inputs.
\end{enumerate}

\subsection{Entanglement swapping}

A \emph{swap at node $i$} using neighbors $y$ and $y'$ (with $i,y,y'$ all distinct) consumes $B$ Bell pairs $[i,y]$ and $B$ Bell pairs $[i,y']$, and produces one Bell pair $[y,y']$. Let
\[
\mu_i^{y,y'}(t)\in\mathbb{Z}_{\ge 0}
\]
be the number of swaps of this kind that node $i$ performs in slot $t$. We take $\mu_i^{y,y'}(t)=\mu_i^{y',y}(t)$.

Swaps have two types of physical feasibility constraints \emph{per slot $t$}:
\begin{itemize}[leftmargin=2em]
\item \textbf{Node budget.} Node $i$ can only do $K_i$ Bell-state measurements (swaps) per time slot:
\begin{equation}
\sum_{\substack{y<y'\\ y,y'\neq i}}\mu_i^{y,y'}(t)\ \le\ K_i,
\qquad \forall i,t.
\label{eq:cap-phys}
\end{equation}

\item \textbf{Availability of Bell pairs.} Node $i$ cannot consume more $[i,y]$ than it physically has (including what is just generated in this slot). Each swap consumes $B$ such pairs. For each $i$ and $y\neq i$,
\begin{equation}
B\sum_{\substack{y'\neq y\\ y'\neq i}}\mu_i^{y,y'}(t)
\ \le\ Q_{iy}(t)+G_{iy}(t).
\label{eq:avail-phys}
\end{equation}
\end{itemize}

After performing all swaps in slot $t$, we can write the \emph{intermediate} inventory,
before serving demand, as
\begin{equation}
\begin{aligned}
Q'_{xy}(t)
&= Q_{xy}(t)\ +\ G_{xy}(t) \\
&\quad + \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\ -\ B\sum_{w\notin\{x,y\}}
\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big) .
\end{aligned}
\label{eq:Qprime-phys}
\end{equation}
Interpretation:
\begin{itemize}[leftmargin=2em]
\item $+\sum_i \mu_i^{x,y}$ are new $[x,y]$ pairs \emph{created} by swaps at third-party nodes $i$.
\item $-B\sum_w(\mu_x^{y,w}+\mu_y^{x,w})$ are $[x,y]$ pairs \emph{consumed} (in blocks of $B$) when $x$ or $y$ performs swaps to create other links.
\end{itemize}

\subsection{Service of consumption demand}

Let $R_{xy}(t)\in\mathbb{Z}_{\ge 0}$ denote how many \emph{consumption requests} between $x,y$ we \emph{serve} in slot $t$. Serving one logical request burns $B$ Bell pairs $[x,y]$.

We can never serve more requests than (i) we have Bell pairs for (after swaps), and (ii) we have demand for. We capture demand using a backlog $D_{xy}(t)$:
\begin{align}
0\ \le\ R_{xy}(t)
&\le \min\Big\{\,\big\lfloor Q'_{xy}(t)/B\big\rfloor,\ D_{xy}(t)+A_{xy}(t)\,\Big\},
\label{eq:R-box}\\
D_{xy}(t{+}1)
&=\big[D_{xy}(t)+A_{xy}(t)-R_{xy}(t)\big]_+ , \quad D_{xy}(0)\ge 0,
\label{eq:D-backlog}\\
Q_{xy}(t{+}1)
&= Q'_{xy}(t) - B R_{xy}(t).
\label{eq:Q-evol}
\end{align}

\subsection{System State.}
State at the beginning of slot $t$: $\{Q_{xy}(t), D_{xy}(t)\}_{xy}$. \\
Decisions in slot $t$: $\{\mu_i^{y,y'}(t)\}$, then $Q'_{xy}(t)$ given by \eqref{eq:Qprime-phys}, then $\{R_{xy}(t)\}$.\\
State evolves by \eqref{eq:D-backlog}, \eqref{eq:Q-evol}.

\section{Average feasibility LP and the capacity region}

We now describe what rates are \emph{in principle} sustainable on average, assuming infinite buffer sizes, in the presence of distillation with cost $B$.

For each pair $xy$, let $r_{xy}\ge 0$ denote the \emph{long-run average} rate at which we successfully serve consumption requests between $x$ and $y$. Let $\sigma_i^{y,y'}\ge 0$ denote the \emph{long-run average} swap rate at node $i$ converting $B$ copies of $[i,y]$ and $B$ copies of $[i,y']$ into one $[y,y']$.

\paragraph{Average resource balance}

Over long time scales, the average rate at which \emph{physical} $[x,y]$ pairs are consumed cannot exceed:
\begin{itemize}[leftmargin=2em]
\item exogenous generation $g_{xy} := \E[G_{xy}(t)]$,
\item plus inflow from swaps that \emph{create} $[x,y]$ (i.e., $\sigma_i^{x,y}$ for $i\notin\{x,y\}$),
\item minus outflow to swaps that \emph{use} $[x,y]$ as an input (i.e., swaps at $x$ or $y$ where $[x,y]$ is consumed in blocks of $B$),
\item minus the service rate $B r_{xy}$ itself.
\end{itemize}

Hence, 
\begin{equation}
B r_{xy}
\ \le\
g_{xy}
+\sum_{i\notin\{x,y\}}\sigma_i^{x,y}
- B\sum_{w\notin\{x,y\}}
\big(\sigma_x^{y,w}+\sigma_y^{x,w}\big).
\label{eq:avg-balance}
\end{equation}
\eqref{eq:avg-balance} says: the average \emph{consumption of Bell pairs} $B r_{xy}$ cannot exceed the average production rate from direct link generation plus net swap inflow.

\paragraph{Average demand satisfaction}

We also require that, in the long run, offered demand is actually served:
\begin{equation}
r_{xy}\ \ge\ \lambda_{xy}.
\label{eq:avg-demand}
\end{equation}

\paragraph{Average node caps}

The per-slot cap \eqref{eq:cap-phys} induces the average constraint
\begin{equation}
\sum_{\substack{y<y'\\ y,y'\neq i}}\sigma_i^{y,y'}
\ \le\ K_i,
\qquad \forall i.
\label{eq:avg-cap}
\end{equation}

\paragraph{Feasibility LP and capacity region}

We summarize the long-term feasibility as:

\begin{align}
&\max_{r,\sigma}  \ 0 && \\
& B r_{xy} \le g_{xy}
+\sum_{i\notin\{x,y\}}\sigma_i^{x,y}
- B\sum_{w\notin\{x,y\}}\big(\sigma_x^{y,w}+\sigma_y^{x,w}\big),
&& \forall xy,
\label{eq:Phard-balance}\\
&r_{xy} \ge \lambda_{xy}, && \forall xy,
\label{eq:Phard-demand}\\
&\sum_{\substack{y<y'\\ y,y'\neq i}}\sigma_i^{y,y'} \le K_i,
&& \forall i,
\label{eq:Phard-cap}\\
&r,\sigma \ge 0. \label{eq:sign}
\end{align}

\begin{definition}[Capacity region $\Lambda^{K,B}$]
Let $\Lambda^{K,B}$ be the set of all rate matrices $\lambda=\{\lambda_{xy}\}_{xy\in\Pairs}$ for which {\rm LP} \eqref{eq:Phard-balance}--\eqref{eq:sign} is feasible. Let $\mathrm{int}(\Lambda^{K,B})$ denote its interior.
\end{definition}
\begin{remark}
If the generation graph with edges $\{xy:\ g_{xy}>0\}$ is disconnected, then for $x,y$ in different components no $r, \sigma$ can make the LP feasible. Hence any $\lambda_{xy}>0$ across components is infeasible. Distillation ($B>1$) shrinks the overall region but does not change this connectivity requirement.
\end{remark}

\section{Dual LP}

We now form a \emph{Lagrangian} for the LP by dualizing the two families of inequalities \eqref{eq:Phard-balance} and \eqref{eq:Phard-demand}, but \emph{not} dualizing the node caps \eqref{eq:Phard-cap}. This split is crucial: it lets us interpret per-node caps as hard constraints, which will later translate directly to local per-slot knapsack problems.

\subsection{Dual variables}

We introduce dual variables:
\begin{itemize}[leftmargin=2em]
\item For each $xy$, let $\alpha_{xy}\ge 0$ be the Lagrange multiplier for the \emph{resource-balance} constraint \eqref{eq:Phard-balance}:
\[
B r_{xy}
-
\Big(
g_{xy}
+\sum_{i\notin\{x,y\}}\sigma_i^{x,y}
- B\sum_{w\notin\{x,y\}}(\sigma_x^{y,w}+\sigma_y^{x,w})
\Big)
\ \le\ 0.
\]

\item For each $xy$, let $\gamma_{xy}\ge 0$ be the Lagrange multiplier for the \emph{demand-satisfaction} constraint \eqref{eq:Phard-demand}:
\[
\lambda_{xy}-r_{xy} \ \le\ 0.
\]
\end{itemize}

\subsection{Lagrangian}

The  Lagrangian is
\begin{equation}
\begin{aligned}
\mathcal{L}(r,\sigma;\alpha,\gamma)
&:= \sum_{xy}\alpha_{xy}\,
\Big(
B r_{xy}
-
g_{xy}
-\sum_{i\notin\{x,y\}}\sigma_i^{x,y}
+ B\sum_{w\notin\{x,y\}}(\sigma_x^{y,w}+\sigma_y^{x,w})
\Big) \\
&\quad
+ \sum_{xy}\gamma_{xy}\,(\lambda_{xy}-r_{xy}),
\end{aligned}
\label{eq:Lagrangian-expand}
\end{equation}
where $r, \sigma$ are subject to (\ref{eq:Phard-cap}) and (\ref{eq:sign}). Expand and collect $r$-terms and $\sigma$-terms:
\begin{equation}
\begin{aligned}
\mathcal{L}(r,\sigma;\alpha,\gamma)
&= \sum_{xy}(B\alpha_{xy}-\gamma_{xy})\,r_{xy} \\
&\quad
+ \sum_i \sum_{y<y'} 
\Big(
B\alpha_{iy}+B\alpha_{iy'}
-\alpha_{yy'}
\Big)\,
\sigma_i^{y,y'} \\
&\quad
-\sum_{xy} \alpha_{xy}\,g_{xy}
+ \sum_{xy}\gamma_{xy}\,\lambda_{xy}.
\end{aligned}
\label{eq:Lagrangian-collected}
\end{equation}

\subsection{Swap weights}

For node $i$ and distinct $y,y'$, define
\begin{equation}
W_i(y,y')
:= \alpha_{yy'}-B\alpha_{iy}-B\alpha_{iy'}.
\label{eq:Wi-def}
\end{equation}
Then the $\sigma$-term in \eqref{eq:Lagrangian-collected} is
\[
\sum_i \sum_{y<y'}
\sigma_i^{y,y'}\,
\big(
B\alpha_{iy}+B\alpha_{iy'}
-\alpha_{yy'}
\big)
= \sum_i \sum_{y<y'} \sigma_i^{y,y'}\,\big(-W_i(y,y')\big).
\]
Equivalently: increasing $\sigma_i^{y,y'}$ is attractive if and only if $W_i(y,y')>0$. Thus $W_i(y,y')$ is exactly the \emph{marginal value} of using a swap at node $i$ to convert $B$ copies each of $[i,y]$ and $[i,y']$ into a single $[y,y']$.

For $B=1$ this reduces to the original form $W_i(y,y')=\alpha_{yy'}-\alpha_{iy}-\alpha_{iy'}$.

\section{From dual to the \emph{online} algorithm}

We now interpret \eqref{eq:Lagrangian-collected} as a per-slot objective. The idea is:
\begin{itemize}[leftmargin=2em]
\item At slot $t$, treat $(\gamma(t),\alpha(t))$ as current dual ``prices'',
\item Choose \emph{instantaneous} actions $(R(t),\mu(t))$ that maximize the per-slot Lagrangian subject to physical feasibility for that slot,
\item Then update $(\gamma,\alpha)$ via a projected subgradient step.
\end{itemize}

\subsection{Per-slot feasible set}

In each slot $t$, the decisions are:
\begin{itemize}[leftmargin=2em]
\item $R_{xy}(t)$, how many \emph{requests} between $x,y$ we serve,
\item $\mu_i^{y,y'}(t)$, how many swaps node $i$ performs across neighbors $y,y'$.
\end{itemize}

The actions must satisfy the per-slot constraints:
\begin{align}
\sum_{\substack{y<y'\\ y,y'\neq i}}\mu_i^{y,y'}(t)
&\le K_i,
&&\forall i,t, \label{eq:slot-cap}\\
B\sum_{\substack{y'\neq y\\ y'\neq i}}\mu_i^{y,y'}(t)
&\le Q_{iy}(t)+G_{iy}(t),
&&\forall i,y\neq i,t, \label{eq:slot-avail}\\
Q'_{xy}(t)
&= Q_{xy}(t)+G_{xy}(t)
+ \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
- B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big),
&&\forall xy,t,
\label{eq:slot-Qprime}\\
0\ \le\ R_{xy}(t)
&\le \min\{\,\lfloor Q'_{xy}(t)/B\rfloor,\ D_{xy}(t)+A_{xy}(t)\,\},
&&\forall xy,t,
\label{eq:slot-Rbox}
\end{align}
and then evolve demand and inventory via \eqref{eq:D-backlog}, \eqref{eq:Q-evol}.

Note that \eqref{eq:slot-cap} and \eqref{eq:slot-avail} are \emph{exactly} the instantaneous versions of the ``hard'' constraints in \eqref{eq:Phard-cap} and physical feasibility.

\subsection{Instantaneous Lagrangian objective}

Using the Lagrangian \eqref{eq:Lagrangian-collected}, the slot-$t$ objective—ignoring constant terms that do not depend on $(R,\mu)$ in that slot—is
\begin{equation}
\begin{aligned}
\mathcal{L}_t(R,\mu;\alpha(t),\gamma(t))
&:= \sum_{xy}
\big(\gamma_{xy}(t)-B\alpha_{xy}(t)\big)\,R_{xy}(t) \\
&\quad + \sum_i \sum_{y<y'}
\mu_i^{y,y'}(t)\,W_i(y,y';t),
\end{aligned}
\label{eq:inst-Lagrangian}
\end{equation}
where $W_i(y,y';t)$ is the (time-varying) weight \eqref{eq:Wi-def} evaluated at the current duals $\alpha(t)$.

The \emph{online backpressure policy} is: at each slot $t$, choose $(R(t),\mu(t))$ to maximize \eqref{eq:inst-Lagrangian} subject to \eqref{eq:slot-cap}–\eqref{eq:slot-Rbox}.

\paragraph{Service decision $R(t)$.}
Fix $xy$. Its contribution to \eqref{eq:inst-Lagrangian} is linear:
\[
(\gamma_{xy}(t)-B\alpha_{xy}(t))\,R_{xy}(t).
\]
Subject to the box \eqref{eq:slot-Rbox}, the optimizer is:
\begin{equation}
R_{xy}(t)\;=\;
\begin{cases}
\min\!\big\{\,\lfloor Q'_{xy}(t)/B\rfloor,\ D_{xy}(t)+A_{xy}(t)\,\big\},
&\text{if }\gamma_{xy}(t)\ge B\alpha_{xy}(t),\\[4pt]
0, &\text{otherwise.}
\end{cases}
\label{eq:Rrule-detailed}
\end{equation}
This has a nice interpretation:
\begin{itemize}[leftmargin=2em]
\item If $\gamma_{xy} \ge B\alpha_{xy}$, unmet demand pressure outweighs $B$ times scarcity pressure, so we serve as much as we physically can.
\item If $\gamma_{xy} < B\alpha_{xy}$, resource scarcity (scaled by $B$) is more ``valuable'' than clearing demand on this edge, so we hold back.
\end{itemize}

\paragraph{Swap decision $\mu(t)$.}
For each node $i$, the contribution to \eqref{eq:inst-Lagrangian} is
\[
\sum_{\substack{y<y'\\ y,y'\neq i}}
\mu_i^{y,y'}(t)\,W_i(y,y';t).
\]
Node $i$ must respect its own physical caps/availability \eqref{eq:slot-cap}--\eqref{eq:slot-avail}. Thus node $i$ solves:
\begin{equation}
\begin{aligned}
\max_{\mu_i^{\cdot,\cdot}(t)\ge 0}\quad
&\sum_{\substack{y<y'\\ y,y'\neq i}}
\mu_i^{y,y'}(t)\,W_i(y,y';t) \\[2pt]
\text{s.t.}\quad
&\sum_{\substack{y<y'\\ y,y'\neq i}}
\mu_i^{y,y'}(t)\ \le\ K_i, \\
&B\sum_{\substack{y'\neq y\\ y'\neq i}}
\mu_i^{y,y'}(t)
\ \le\ Q_{iy}(t)+G_{iy}(t),
\qquad \forall y\neq i.
\end{aligned}
\label{eq:nodeILP-detailed}
\end{equation}
The solution to the above optimization is as follows: Each node $i$ allocates up to $K_i$ swaps, subject to not over-consuming any $[i,y]$ link (in blocks of $B$), to the
largest positive $W_i(y,y')$. If all $W_i(y,y')\le 0$, the optimum is $\mu_i^{y,y'}(t)=0$.

\section{Queue updates}

We now show how $\gamma(t)$ and $\alpha(t)$ evolve. It establishes that the two “queues” are in fact projected subgradient ascent iterates on the dual problem (this is well known from stochastic primal-dual literature). 

Define the \emph{per-slot residuals} for the two dualized constraints:

\paragraph{Demand residual.}
Constraint \eqref{eq:Phard-demand} is $r_{xy} \ge \lambda_{xy}$.
In a single slot we interpret $r_{xy}(t)=R_{xy}(t)$ and $\lambda_{xy}=A_{xy}(t)$. The \emph{violation} is
\[
h^{\rm dem}_{xy}(t) := A_{xy}(t)-R_{xy}(t).
\]

\paragraph{Resource residual.}
Constraint \eqref{eq:Phard-balance} is
\[
B r_{xy} - 
\Big(
g_{xy}
+\sum_{i\notin\{x,y\}}\sigma_i^{x,y}
- B\sum_{w\notin\{x,y\}}(\sigma_x^{y,w}+\sigma_y^{x,w})
\Big)
\ \le\ 0.
\]
Per slot we interpret $r_{xy}(t)=R_{xy}(t)$, $\sigma_i^{y,y'}(t)=\mu_i^{y,y'}(t)$, and $g_{xy}(t)=G_{xy}(t)$, so the \emph{violation} is
\begin{equation}
\begin{aligned}
h^{\rm res}_{xy}(t)
&:= B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)\\
&\quad
- \Big(
G_{xy}(t)
+\sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\Big).
\end{aligned}
\label{eq:hres-expand}
\end{equation}

Standard primal--dual control (a.k.a.\ stochastic subgradient ascent on the dual with
unit steps and projection onto $\mathbb{R}_+^{|\Pairs|}$) gives:
\begin{align}
\gamma_{xy}(t{+}1)
&=\Big[\,
\gamma_{xy}(t)+h^{\rm dem}_{xy}(t)
\,\Big]_+
=\Big[\,
\gamma_{xy}(t)+A_{xy}(t)-R_{xy}(t)
\,\Big]_+,
\label{eq:gamma-update-full}\\[4pt]
\alpha_{xy}(t{+}1)
&=\Big[\,
\alpha_{xy}(t)+h^{\rm res}_{xy}(t)
\,\Big]_+\nonumber\\
&=\Big[\,
\alpha_{xy}(t)
+ B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- \Big(
G_{xy}(t)
+\sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\Big)
\,\Big]_+.
\label{eq:alphares-update-full}
\end{align}

\paragraph{Important observation.}
The updates \eqref{eq:gamma-update-full} and \eqref{eq:alphares-update-full} are \emph{exactly} the physical queue-like recursions:
\begin{itemize}[leftmargin=2em]
\item $\gamma_{xy}(t)$ behaves like the demand backlog $D_{xy}(t)$ (compare \eqref{eq:gamma-update-full} to \eqref{eq:D-backlog}).
\item $\alpha_{xy}(t)$ measures scarcity of $[x,y]$ pairs as inputs to swaps and service: it increases when we \emph{consume} $[x,y]$ via $R$ or via swaps at $x$ or $y$ (each by $B$ units), and it decreases when we \emph{produce} $[x,y]$ via $G_{xy}$ or via swaps at third-party nodes.
\end{itemize}

\section{Stability via Lyapunov analysis}

We now prove that if the arrival rates $\lambda$ are strictly inside the capacity region $\Lambda^{K,B}$, then under the online policy described above, both $\gamma(t)$ and $\alpha(t)$ remain strongly stable, and the long-run service rate meets (and slightly exceeds) demand.

\subsection{Lyapunov function and drift definition}

Define the quadratic Lyapunov function
\begin{equation}
L(\gamma,\alpha)
:= \frac12 \sum_{xy}
\Big(
\gamma_{xy}^2 + (\alpha_{xy})^2
\Big).
\label{eq:Lyap-def}
\end{equation}
Define the conditional drift
\begin{equation}
\Delta(t)
:= \E\!\left[
L(\gamma(t{+}1),\alpha(t{+}1))
-
L(\gamma(t),\alpha(t))
\ \big|\ \gamma(t),\alpha(t)
\right].
\label{eq:drift-def}
\end{equation}

We will upper bound $\Delta(t)$ by an affine function of $\gamma(t)$ and $\alpha(t)$; if $\lambda\in\mathrm{int}(\Lambda^{K,B})$ we will get a \emph{negative drift outside a bounded set}.

\subsection{Bounding the one-step drift}

We start from the generic inequality, for any scalar $x$ and increment $y$:
\[
([x{+}y]_+)^2 - x^2
\ \le\ y^2 + 2xy.
\]
Apply this to each coordinate of \eqref{eq:gamma-update-full} and sum:
\begin{equation}
\begin{aligned}
\gamma_{xy}(t{+}1)^2 - \gamma_{xy}(t)^2
&\le
\big(A_{xy}(t)-R_{xy}(t)\big)^2
+ 2\,\gamma_{xy}(t)\,\big(A_{xy}(t)-R_{xy}(t)\big).
\end{aligned}
\label{eq:gamma-square-step}
\end{equation}
Similarly, apply it to each coordinate of \eqref{eq:alphares-update-full}:
\begin{equation}
\begin{aligned}
\alpha_{xy}(t{+}1)^2
- \alpha_{xy}(t)^2
&\le
\Big(
B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- G_{xy}(t) \\
&\qquad
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\Big)^2 \\
&\quad
+ 2\,\alpha_{xy}(t)\Big[
B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- G_{xy}(t) \\
&\qquad
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\Big].
\end{aligned}
\label{eq:alphares-square-step}
\end{equation}

Sum \eqref{eq:gamma-square-step}+\eqref{eq:alphares-square-step} over all $xy$, divide by $2$, and then take conditional expectation given $(\gamma(t),\alpha(t))$.
Because $A_{xy}(t)$ and $G_{xy}(t)$ have bounded second moments, $R_{xy}(t)$ and $\mu_i^{y,y'}(t)$ are bounded by the hard caps \eqref{eq:slot-cap} and inventories \eqref{eq:slot-avail}, \emph{all} the quadratic terms can be absorbed into a uniform constant $C<\infty$. We obtain:

\begin{equation}
\begin{aligned}
\Delta(t)\ \le\ &
C\\
&+ \sum_{xy} \gamma_{xy}(t)\,
\E\!\left[ A_{xy}(t)-R_{xy}(t)\ \big|\ \gamma(t),\alpha(t)\right] \\
&+ \sum_{xy} \alpha_{xy}(t)\,
\E\!\Big[
B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- G_{xy}(t) \\
&\qquad\qquad\qquad
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\ \Big|\ \gamma(t),\alpha(t)\Big].
\end{aligned}
\label{eq:drift-before-rearrange}
\end{equation}

Now use $\E[A_{xy}(t)]=\lambda_{xy}$ and $\E[G_{xy}(t)]=g_{xy}$:
\begin{equation}
\begin{aligned}
\Delta(t)\ \le\ &
C
+ \sum_{xy} \gamma_{xy}(t)\,
\big(\lambda_{xy} - \E[R_{xy}(t)\mid\cdot]\big) \\
&+ \sum_{xy} \alpha_{xy}(t)\,
\E\!\Big[
B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\ \Big|\ \cdot\Big]\\
&- \sum_{xy} \alpha_{xy}(t)\,g_{xy},
\end{aligned}
\label{eq:drift-rearrange}
\end{equation}
where $\cdot$ denotes conditioning on $(\gamma(t),\alpha(t))$.

Note that
\[
\sum_{xy} \alpha_{xy}(t)
\Big(
B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\Big)
= - \sum_i \sum_{y<y'}
\mu_i^{y,y'}(t)\,W_i(y,y';t),
\]
where $W_i$ is from \eqref{eq:Wi-def}. This identity is standard: each unit of $\mu_i^{y,y'}$ \emph{increases} $\alpha_{yy'}$ (output link) and \emph{decreases} $\alpha_{iy},\alpha_{iy'}$ (inputs, weighted by $B$), giving exactly $-W_i(y,y')$.

Plugging in:
\begin{equation}
\begin{aligned}
\Delta(t)\ \le\ &
C
+ \sum_{xy} \gamma_{xy}(t)\,
\big(\lambda_{xy} - \E[R_{xy}(t)\mid\cdot]\big) \\
&+ B\sum_{xy} \alpha_{xy}(t)\,
\E[R_{xy}(t)\mid\cdot]
- \sum_i \E\!\Big[\sum_{y<y'}\mu_i^{y,y'}(t)\,W_i(y,y';t)\ \Big|\ \cdot\Big] \\
&- \sum_{xy} \alpha_{xy}(t)\,g_{xy}.
\end{aligned}
\label{eq:drift-node-form}
\end{equation}

\subsection{Max-weight (backpressure) dominance}

Let $\tilde\pi$ be an \emph{arbitrary} alternative feasible policy that obeys the same per-slot physical constraints \eqref{eq:slot-cap}--\eqref{eq:slot-Rbox}. By construction,
our online rule $(R(t),\mu(t))$ \emph{maximizes} the instantaneous Lagrangian \eqref{eq:inst-Lagrangian} over that same feasible set. Therefore:
\begin{align}
\sum_{xy} (\gamma_{xy}(t)-B\alpha_{xy}(t))
\E\!\left[R_{xy}(t)\mid\cdot\right]
&\ge
\sum_{xy} (\gamma_{xy}(t)-B\alpha_{xy}(t))
\E\!\left[R_{xy}^{\tilde\pi}(t)\mid\cdot\right],
\label{eq:maxweight-service}\\
\sum_i \E\!\left[\sum_{y<y'} \mu_i^{y,y'}(t)\,W_i(y,y';t)\mid\cdot\right]
&\ge
\sum_i \E\!\left[\sum_{y<y'} (\mu_i^{y,y'})^{\tilde\pi}(t)\,W_i(y,y';t)\mid\cdot\right].
\label{eq:maxweight-swap}
\end{align}

Define the instantaneous Lagrangian value
\[
\mathcal{L}_t(R,\mu) :=
\sum_{xy} (\gamma_{xy}(t)-B\alpha_{xy}(t)) R_{xy}
+ \sum_i \sum_{y<y'} \mu_i^{y,y'} W_i(y,y';t).
\]
Then \eqref{eq:maxweight-service}--\eqref{eq:maxweight-swap} say that our policy maximizes $\mathcal{L}_t$ among all feasible $(R,\mu)$ in slot $t$.

\subsection{Slack policy under interior load}

If $\lambda\in\mathrm{int}(\Lambda^{K,B})$, then there exists $\epsilon>0$ and a feasible $(r,\sigma)$ for the LP such that
\[
r_{xy} \ge \lambda_{xy}+\epsilon,
\]
and
\[
B r_{xy} \le g_{xy}
+\sum_{i\notin\{x,y\}}\sigma_i^{x,y}
- B\sum_{w\notin\{x,y\}}(\sigma_x^{y,w}+\sigma_y^{x,w}),
\]
with $\sum_{y<y'}\sigma_i^{y,y'}\le K_i$.
By standard time-sharing arguments, we can build a \emph{stationary randomized per-slot policy} $\pi^\star$ that achieves per-slot expectations
\[
\E_{\pi^\star}[R_{xy}(t)] = r_{xy},\qquad
\E_{\pi^\star}[\mu_i^{y,y'}(t)] = \sigma_i^{y,y'},
\]
while obeying \eqref{eq:slot-cap}--\eqref{eq:slot-Rbox} almost surely.

Under $\pi^\star$ we therefore have
\begin{align}
\E_{\pi^\star}[R_{xy}(t)] &\ge \lambda_{xy}+\epsilon, \label{eq:piR-ineq}\\
\E_{\pi^\star}\!\left[
B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\right]
&\le
g_{xy}, \label{eq:pi-balance-ineq}
\end{align}
for all $xy$.

\subsection{Final drift inequality}

Plug $\tilde\pi=\pi^\star$ into \eqref{eq:maxweight-service}--\eqref{eq:maxweight-swap}, then substitute into \eqref{eq:drift-node-form}, using that our policy maximizes $\mathcal{L}_t$ so its contribution is no larger than that of $\pi^\star$ when we carry a minus sign. We obtain:

\begin{equation}
\begin{aligned}
\Delta(t)
&\le C
+ \sum_{xy} \gamma_{xy}(t)\,
\big(\lambda_{xy} - \E_{\pi^\star}[R_{xy}(t)]\big)
\\
&\quad
+ \sum_{xy} \alpha_{xy}(t)\,
\E_{\pi^\star}\!\Big[
B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\Big]
- \sum_{xy} \alpha_{xy}(t) g_{xy}.
\end{aligned}
\label{eq:drift-pre-simplify}
\end{equation}

Now apply \eqref{eq:piR-ineq} and \eqref{eq:pi-balance-ineq}:
\begin{align*}
\lambda_{xy} - \E_{\pi^\star}[R_{xy}(t)]
&\le -\epsilon,\\
\E_{\pi^\star}\!\Big[
B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)
- \sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)
\Big]
- g_{xy}
&\le 0.
\end{align*}
Therefore
\begin{equation}
\Delta(t)
\ \le\
C
-\epsilon\,\sum_{xy}\gamma_{xy}(t).
\label{eq:final-drift}
\end{equation}

\subsection{Stability and throughput optimality}

Inequality \eqref{eq:final-drift} says: outside a bounded set (where $\sum\gamma_{xy}$ is large), the expected Lyapunov drift is strictly negative. By a standard Foster--Lyapunov argument for Markov chains with countable state (or the stochastic network stability framework), this implies \emph{strong stability} of $\gamma(t)$, i.e.,
\[
\limsup_{T\to\infty}\frac{1}{T}\sum_{t=0}^{T-1}\sum_{xy}\E[\gamma_{xy}(t)]
\ <\ \infty.
\]
Because $\alpha(t)$ is also in the Lyapunov function \eqref{eq:Lyap-def} and its contribution to $\Delta(t)$ is nonpositive in \eqref{eq:final-drift}, we similarly get strong stability of $\alpha(t)$.

Finally, rearranging the queue update \eqref{eq:gamma-update-full} in steady state and summing over $t$ gives
\[
\frac{1}{T}\sum_{t=0}^{T-1}\E[R_{xy}(t)]
\ \ge\
\lambda_{xy}
- \frac{\E[\gamma_{xy}(T)]-\E[\gamma_{xy}(0)]}{T}.
\]
Taking $T\to\infty$ and using stability of $\gamma$ shows
\[
\liminf_{T\to\infty}
\frac{1}{T}\sum_{t=0}^{T-1}\E[R_{xy}(t)]
\ \ge\ \lambda_{xy},
\qquad \forall xy.
\]

We summarize this as our main result.

\begin{theorem}[Throughput optimality with distillation and hard per-node swap caps]
\label{thm:main}
Suppose $\lambda\in\mathrm{int}(\Lambda^{K,B})$. Consider the following distributed policy:

\begin{enumerate}[leftmargin=2em,label=(\roman*)]
\item \textbf{At each slot $t$:} compute the weights
\[
W_i(y,y';t)=\alpha_{yy'}(t)-B\alpha_{iy}(t)-B\alpha_{iy'}(t).
\]

\item \textbf{Swaps at node $i$:} choose $\mu_i^{y,y'}(t)$ to solve the local max-weight linear program \eqref{eq:nodeILP-detailed} subject to the instantaneous hard cap \eqref{eq:slot-cap} and availability constraints \eqref{eq:slot-avail}.

\item \textbf{Service on pair $xy$:} choose $R_{xy}(t)$ according to \eqref{eq:Rrule-detailed}.

\item \textbf{Update dual queues:}
\begin{align*}
\gamma_{xy}(t{+}1)
&=\big[\gamma_{xy}(t)+A_{xy}(t)-R_{xy}(t)\big]_+,\\
\alpha_{xy}(t{+}1)
&=\Big[\alpha_{xy}(t)
+ B R_{xy}(t)
+ B\sum_{w\notin\{x,y\}}\big(\mu_x^{y,w}(t)+\mu_y^{x,w}(t)\big)\\
&\qquad\qquad
- \Big(G_{xy}(t)+\sum_{i\notin\{x,y\}}\mu_i^{x,y}(t)\Big)
\Big]_+.
\end{align*}
\end{enumerate}

Then both queues $\gamma(t)$ and $\alpha(t)$ are strongly stable, and the long-run average service rate meets the demand rate:
\[
\liminf_{T\to\infty}
\frac1T \sum_{t=0}^{T-1}\E[R_{xy}(t)]
\ \ge\ \lambda_{xy},
\quad \forall xy\in\Pairs.
\]
\end{theorem}

\section*{Discussion}

\begin{remark}[Why two queues are necessary]
The queue $\gamma_{xy}(t)$ acts like an ordinary demand backlog: it grows when arrivals $A_{xy}(t)$ exceed service $R_{xy}(t)$. The queue $\alpha_{xy}(t)$ is \emph{not} physical backlog but a scarcity shadow price: it grows when link $[x,y]$ is ``over-used'' as an input to swaps or for service (each such use now costs $B$ Bell pairs), compared to how much it is replenished by generation or swaps \emph{into} $[x,y]$. The combination enforces both ``serve requests'' and ``don't overconsume scarce entanglement,'' taking distillation overhead into account.
\end{remark}

\begin{remark}[Decentralization]
Node $i$ needs:
\begin{itemize}[leftmargin=2em]
\item $\alpha_{iy}(t)$ for every neighbor $y$ to price consuming $[i,y]$ as a swap input (with cost $B$),
\item $\alpha_{yy'}(t)$ for $(y,y')$ pairs it can create.
\end{itemize}
This information can be exchanged locally (gossip). Each node then solves \eqref{eq:nodeILP-detailed} independently.
\end{remark}

\begin{remark}[Effect of $B$ on the capacity region]
As $B$ increases, each logical request and swap consumes more physical Bell pairs, strictly shrinking $\Lambda^{K,B}$. For fixed $g_{xy}$ and $K_i$, the maximum sustainable end-to-end rates scale roughly like $1/B$ in simple symmetric topologies.
\end{remark}
